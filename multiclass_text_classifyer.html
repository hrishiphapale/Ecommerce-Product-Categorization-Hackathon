def text_clean(content):
    # Lowercase the content
    content = content.lower()

    # Remove punctuation
    content = re.sub('[()!?]', '', content)  # Replace punctuation with a space

    # Remove non-alphanumeric occurrences
    content = re.sub("[^a-z0-9 ]", "", content)  # Replace non-alphanumeric characters with a space

    # Remove the @mention
    content = re.sub("@[A-Za-z0-9]+", "", content)  

    # Remove the hashtags
    content = re.sub("#[A-Za-z0-9]+", "", content)  

    # Remove any Links
    content = re.sub(r"http\S+", "", content)  
    content = re.sub(r"www.\S+", "", content)  

    return content
